Code Examples

// npm install phone-validation-with-flag
import { PhoneInput } from 'phone-validation-with-flag';

const phone = new PhoneInput('#my-phone', {
  defaultCountry: 'US',
  placeholder: 'Phone number',
  theme: 'auto',       // 'auto' | 'light' | 'dark'
  onChange: (val) => {
    console.log(val.full);       // '+923001234567'
    console.log(val.dialCode);   // '+92'
    console.log(val.number);     // '3001234567'
    console.log(val.countryCode);// 'PK'
  },
});

// Programmatic API
phone.getValue();             // → { countryCode, dialCode, number, full }
phone.setValue('GB', '7911123456');
phone.reset();               // clear to default
